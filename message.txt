-- Farm logic
local farming = false
local coinsCollected = 0
local MAX_BAG = 40

local function getCoins()
    for _, map in pairs(workspace:GetChildren()) do
        if map:FindFirstChild("CoinContainer") then
            return map.CoinContainer
        end
    end
end

local function findNearestCoin()
    local character = LocalPlayer.Character
    if not character then return nil end
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    local container = getCoins()
    if not container then return nil end

    local nearest, dist = nil, math.huge
    for _, coin in pairs(container:GetChildren()) do
        if coin:IsA("BasePart") and coin.Parent and coin:IsDescendantOf(workspace) then
            local d = (hrp.Position - coin.Position).Magnitude
            if d < dist then
                dist = d
                nearest = coin
            end
        end
    end
    return nearest
end

local function moveToCoin(coin)
    local character = LocalPlayer.Character
    if not character then return end
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp or not coin or not coin.Parent then return end

    local distance = (hrp.Position - coin.Position).Magnitude
    local speed = 20
    local time = math.clamp(distance / speed, 0.3, 3)

    local tween = TweenService:Create(hrp, TweenInfo.new(time, Enum.EasingStyle.Linear), {CFrame = coin.CFrame + Vector3.new(0,3,0)})
    local finished = false
    tween.Completed:Connect(function()
        finished = true
        if coin and coin.Parent then
            coin:Destroy()
            coinsCollected = coinsCollected + 1
            bagStatus.Text = "Balo: "..coinsCollected.."/"..MAX_BAG
            if coinsCollected >= MAX_BAG then
                print("Balo đầy! Reset nhân vật...")
                if LocalPlayer.Character then
                    LocalPlayer.Character:BreakJoints() -- reset nhân vật
                end
                coinsCollected = 0
                bagStatus.Text = "Balo: 0/"..MAX_BAG
            end
        end
    end)
    tween:Play()

    repeat task.wait() until finished
end

local function autoFarm()
    while farming do
        local coin = findNearestCoin()
        if coin then
            moveToCoin(coin)
        else
            task.wait(0.1)
        end
    end
end

toggleBtn.MouseButton1Click:Connect(function()
    farming = not farming
    if farming then
        toggleBtn.Text = "Stop Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 80, 80)
        status.Text = "Status: ON"
        status.TextColor3 = Color3.fromRGB(100,255,100)
        task.spawn(autoFarm)
    else
        toggleBtn.Text = "Start Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(80,160,80)
        status.Text = "Status: OFF"
        status.TextColor3 = Color3.fromRGB(255,100,100)
    end
end)
