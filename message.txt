-- AutoFarm GUI Mobile với nút Close
local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local LocalPlayer = Players.LocalPlayer

if LocalPlayer.PlayerGui:FindFirstChild("MiniAutoFarmGUI") then
    LocalPlayer.PlayerGui.MiniAutoFarmGUI:Destroy()
end

local gui = Instance.new("ScreenGui")
gui.Name = "MiniAutoFarmGUI"
gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
gui.ResetOnSpawn = false

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 220, 0, 140)
frame.Position = UDim2.new(1, -230, 1, -150)
frame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
frame.BorderSizePixel = 0
frame.Active = true
frame.Parent = gui

-- Tiêu đề
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, -30, 0, 30)
title.BackgroundColor3 = Color3.fromRGB(55, 55, 55)
title.Text = "AutoFarm (kéo ở đây)"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Font = Enum.Font.SourceSansBold

title.Parent = frame

-- Nút Close [X]
local closeBtn = Instance.new("TextButton")

closeBtn.Position = UDim2.new(1, -30, 0, 0)

    gui:Destroy()

-- Trạng thái
local status = Instance.new("TextLabel")
status.Size = UDim2.new(1, -20, 0, 30)
status.Position = UDim2.new(0, 10, 0, 100)
status.BackgroundTransparency = 1
status.Text = "Status: OFF"
status.TextColor3 = Color3.fromRGB(255, 120, 120)
status.Font = Enum.Font.SourceSans
status.TextSize = 16
status.Parent = frame

-- Farm logic
local farming = false
local cooldown = 0.3

local function farmCoins()
    while farming do
        local normal = workspace:FindFirstChild("Normal")
        if normal and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local container = normal:FindFirstChild("CoinContainer")
            if container then
                for _, coin in pairs(container:GetChildren()) do
                    if coin:IsA("BasePart") and farming then
                        LocalPlayer.Character:PivotTo(coin.CFrame)
                        task.wait(cooldown)
                    end
                end
            end
        end
        task.wait()
    end
end

toggleBtn.MouseButton1Click:Connect(function()
    farming = not farming
    if farming then
        toggleBtn.Text = "Stop"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200,80,80)
        status.Text = "Status: ON"
        status.TextColor3 = Color3.fromRGB(120, 255, 120)
        task.spawn(farmCoins)
    else
        toggleBtn.Text = "Start"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(80,160,80)
        status.Text = "Status: OFF"
        status.TextColor3 = Color3.fromRGB(255,120,120)
    end
end)

-- Kéo GUI cho mobile
do
    local dragging, dragInput, dragStart, startPos

    local function update(input)
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end

    title.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
            update(input)
        end
    end)
end
