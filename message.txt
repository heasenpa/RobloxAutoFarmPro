-- AutoFarm với GUI (hiện chắc chắn)
local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local LocalPlayer = Players.LocalPlayer or Players:GetPropertyChangedSignal("LocalPlayer"):Wait()

-- Xóa GUI cũ nếu có
pcall(function()
    if game:GetService("CoreGui"):FindFirstChild("FarmGUI") then
        game:GetService("CoreGui").FarmGUI:Destroy()
    end
end)

-- GUI nhỏ gọn
local gui = Instance.new("ScreenGui")

status.Font = Enum.Font.SourceSans
status.TextSize = 16

-- Farm logic
local farming = false

local function getCoinContainer()
    for _, map in pairs(workspace:GetChildren()) do
        if map:FindFirstChild("CoinContainer") then
            return map.CoinContainer
        end
    end
end

local function getNearestCoin()
    local container = getCoinContainer()
    if not container then return nil end

    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local hrp = char.HumanoidRootPart

    local nearest, minDist = nil, math.huge
    for _, coin in pairs(container:GetChildren()) do
        if coin:IsA("BasePart") and coin.Parent then
            local dist = (hrp.Position - coin.Position).Magnitude
            if dist < minDist then
                minDist = dist
                nearest = coin
            end
        end
    end
    return nearest
end

local function moveToCoin(coin)
    local char = LocalPlayer.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local distance = (hrp.Position - coin.Position).Magnitude
    local speed = 25
    local time = math.clamp(distance / speed, 0.4, 4)

    local tween = TweenService:Create(hrp, TweenInfo.new(time, Enum.EasingStyle.Linear), {
        CFrame = coin.CFrame + Vector3.new(0,3,0)
    })
    tween:Play()
    tween.Completed:Wait()
end

local function autoFarm()
    while farming do
        local coin = getNearestCoin()
        if coin then
            moveToCoin(coin)
        else
            task.wait(0.5)
        end
    end
end

toggleBtn.MouseButton1Click:Connect(function()
    farming = not farming
    if farming then
        toggleBtn.Text = "Stop Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 80, 80)
        status.Text = "Status: ON"
        status.TextColor3 = Color3.fromRGB(100,255,100)
        task.spawn(autoFarm)
    else
        toggleBtn.Text = "Start Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(80,160,80)
        status.Text = "Status: OFF"
        status.TextColor3 = Color3.fromRGB(255,100,100)
    end
end)
