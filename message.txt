-- AutoFarm Tween Part (Codex Mobile Friendly)
local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local LocalPlayer = Players.LocalPlayer

-- Xóa GUI cũ nếu có
if LocalPlayer.PlayerGui:FindFirstChild("FarmGUI") then
    LocalPlayer.PlayerGui.FarmGUI:Destroy()
end

-- GUI
local gui = Instance.new("ScreenGui")
gui.Name = "FarmGUI"
gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
gui.ResetOnSpawn = false

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 180, 0, 100)
frame.Position = UDim2.new(0, 10, 0, 250)
frame.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
frame.BorderSizePixel = 0
frame.Parent = gui

local toggleBtn = Instance.new("TextButton")
toggleBtn.Size = UDim2.new(1, -20, 0, 40)
toggleBtn.Position = UDim2.new(0, 10, 0, 10)
toggleBtn.BackgroundColor3 = Color3.fromRGB(80, 160, 80)
toggleBtn.Text = "Start Farm"
toggleBtn.TextColor3 = Color3.fromRGB(255,255,255)
toggleBtn.Font = Enum.Font.SourceSansBold
toggleBtn.TextSize = 18
toggleBtn.Parent = frame

local status = Instance.new("TextLabel")
status.Size = UDim2.new(1, -20, 0, 30)
status.Position = UDim2.new(0, 10, 0, 60)
status.BackgroundTransparency = 1
status.Text = "Status: OFF"
status.TextColor3 = Color3.fromRGB(255,100,100)
status.Font = Enum.Font.SourceSans
status.TextSize = 16
status.Parent = frame

-- FlyPart vô hình
local flyPart = Instance.new("Part")
flyPart.Anchored = true
flyPart.CanCollide = false
flyPart.Transparency = 1
flyPart.Size = Vector3.new(6,1,6)
flyPart.Name = "FlyPart"
flyPart.Parent = workspace

-- Weld nhân vật vào flyPart
local function weldToPart(char)
    task.wait(1)
    if char and char:FindFirstChild("HumanoidRootPart") then
        local hrp = char.HumanoidRootPart
        flyPart.CFrame = hrp.CFrame
        local weld = Instance.new("WeldConstraint")
        weld.Part0 = hrp
        weld.Part1 = flyPart
        weld.Parent = hrp
    end
end

weldToPart(LocalPlayer.Character)
LocalPlayer.CharacterAdded:Connect(weldToPart)

-- Farm logic
local farming = false

local function getCoins()
    for _, map in pairs(workspace:GetChildren()) do
        if map:FindFirstChild("CoinContainer") then
            return map.CoinContainer
        end
    end
end

local function flyTo(targetPos)
    local distance = (flyPart.Position - targetPos).Magnitude
    local time = distance / 60 -- chỉnh tốc độ bay
    local tween = TweenService:Create(
        flyPart,
        TweenInfo.new(time, Enum.EasingStyle.Linear),
        {Position = targetPos + Vector3.new(0,3,0)} -- bay cao hơn coin 1 chút
    )
    tween:Play()
    tween.Completed:Wait()
end

local function autoFarm()
    while farming do
        local container = getCoins()
        if container then
            local coins = container:GetChildren()
            if #coins > 0 then
                local nearest, dist = nil, math.huge
                for _, coin in pairs(coins) do
                    if coin:IsA("BasePart") then
                        local d = (flyPart.Position - coin.Position).Magnitude
                        if d < dist then
                            nearest, dist = coin, d
                        end
                    end
                end
                if nearest then
                    flyTo(nearest.Position)
                end
            end
        end
        task.wait(0.1)
    end
end

toggleBtn.MouseButton1Click:Connect(function()
    farming = not farming
    if farming then
        toggleBtn.Text = "Stop Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 80, 80)
        status.Text = "Status: ON"
        status.TextColor3 = Color3.fromRGB(100,255,100)
        task.spawn(autoFarm)
    else
        toggleBtn.Text = "Start Farm"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(80,160,80)
        status.Text = "Status: OFF"
        status.TextColor3 = Color3.fromRGB(255,100,100)
    end
end)
